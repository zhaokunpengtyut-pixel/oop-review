**4.1** 阅读以下程序，写出运行结果。

```cpp
#include <iostream>
using namespace std;

class Base {
protected:
    int a;
public:
    Base(int x) : a(x) {}
    void show() { cout << "Base a=" << a << endl; }
};

class Derived : public Base {
private:
    int b;
public:
    Derived(int x, int y) : Base(x), b(y) {}
    void show() {
        cout << "Derived a=" << a << ", b=" << b << endl;
    }
};

int main() {
    Base base(10);
    Derived derived(20, 30);
    base.show();
    derived.show();
    return 0;
}
```
